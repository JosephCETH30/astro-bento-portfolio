Visualisasi Data dengan Chart.js

Fungsi ini digunakan untuk membuat visualisasi data keuangan menggunakan Chart.js. Fungsi ini menghasilkan dua jenis chart: bar chart untuk perbandingan pendapatan dan pengeluaran, serta pie chart untuk rincian pengeluaran. 

function createCharts(income, fixedExpenses, variableExpenses, savings) {
    const ctx = document.getElementById('barChart').getContext('2d');
    const pieCtx = document.getElementById('pieChart').getContext('2d');

    // Bar Chart
    barChart = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: ['Pendapatan', 'Pengeluaran Tetap', 'Pengeluaran Variabel', 'Tabungan'],
            datasets: [{
                label: 'Jumlah',
                data: [income, fixedExpenses, variableExpenses, savings],
                backgroundColor: [
                    'rgba(16, 185, 129, 0.8)', // Hijau
                    'rgba(239, 68, 68, 0.8)',  // Merah
                    'rgba(245, 158, 11, 0.8)', // Oranye
                    'rgba(59, 130, 246, 0.8)'  // Biru
                ],
                borderWidth: 2,
                borderRadius: 8,
                barThickness: 40,
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
                tooltip: {
                    backgroundColor: 'rgba(30, 41, 59, 0.9)',
                    titleColor: '#F8FAFC',
                    bodyColor: '#F8FAFC',
                    borderColor: 'rgba(255, 255, 255, 0.1)',
                    borderWidth: 1,
                    padding: 12,
                    cornerRadius: 8,
                    callbacks: {
                        label: function(context) {
                            return formatCurrency(context.raw);
                        }
                    }
                }
            }
        }
    });

    // Pie Chart
    pieChart = new Chart(pieCtx, {
        type: 'doughnut',
        data: {
            labels: ['Pengeluaran Tetap', 'Pengeluaran Variabel', 'Tabungan'],
            datasets: [{
                data: [fixedExpenses, variableExpenses, savings],
                backgroundColor: [
                    'rgba(239, 68, 68, 0.8)',
                    'rgba(245, 158, 11, 0.8)',
                    'rgba(59, 130, 246, 0.8)'
                ],
                borderWidth: 2,
                borderRadius: 8,
                hoverOffset: 4
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            cutout: '70%',
            plugins: {
                legend: {
                    position: 'bottom',
                    labels: {
                        color: '#94A3B8',
                        font: {
                            family: 'Satoshi',
                            size: 12
                        },
                        padding: 20,
                        usePointStyle: true,
                        pointStyle: 'circle'
                    }
                },
                tooltip: {
                    backgroundColor: 'rgba(30, 41, 59, 0.9)',
                    titleColor: '#F8FAFC',
                    bodyColor: '#F8FAFC',
                    borderColor: 'rgba(255, 255, 255, 0.1)',
                    borderWidth: 1,
                    padding: 12,
                    cornerRadius: 8,
                    callbacks: {
                        label: function(context) {
                            const value = context.raw;
                            const total = context.dataset.data.reduce((a, b) => a + b, 0);
                            const percentage = ((value / total) * 100).toFixed(1);
                            return `${context.label}: ${formatCurrency(value)} (${percentage}%)`;
                        }
                    }
                }
            }
        }
    });
}

Fitur Utama:

  • Membuat bar chart untuk membandingkan pendapatan, pengeluaran, dan tabungan
  • Membuat pie chart untuk melihat proporsi pengeluaran
  • Tooltip yang informatif dengan format mata uang
  • Desain responsif dan modern
  • Warna yang konsisten dan mudah dibedakan

Parameter:

  • income: Total pendapatan
  • fixedExpenses: Total pengeluaran tetap
  • variableExpenses: Total pengeluaran variabel
  • savings: Total tabungan

Integrasi Asisten AI

// Contoh penggunaan Asisten AI
async function askAI(question) {
  const response = await fetch('https://api.yourbudgetapp.com/ai-assistant', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({ question })
  });
  const data = await response.json();
  return data.answer;
}

askAI("Bagaimana cara menabung lebih banyak bulan ini?").then(answer => {
  console.log("Asisten AI:", answer);
});

Memulai dengan Proyek

Untuk memulai dengan proyek Budgy, ikuti langkah-langkah berikut untuk mengkloning dan menyiapkan repositori: 

# Kloning repositori
git clone https://github.com/JosephCETH30/budgy.git

# Masuk ke direktori proyek
cd budgy

# Buka proyek di editor kode pilihan Anda
code .

# Instal dependensi (jika menggunakan npm)
npm install

# Mulai server pengembangan
npm run dev

Persyaratan:

  • Git terinstal di sistem Anda
  • Node.js dan npm (untuk pengembangan)
  • Editor kode (VS Code direkomendasikan)

FAQ

Bagaimana cara kerja kalkulator tabungan?

Kalkulator tabungan membantu Anda menentukan berapa banyak yang dapat Anda tabung berdasarkan pendapatan, pengeluaran, dan tujuan keuangan Anda. Cukup masukkan pendapatan dan pengeluaran bulanan Anda, dan kalkulator akan menunjukkan skenario tabungan potensial dan rekomendasi.

Saran apa yang dapat diberikan oleh asisten AI?

Asisten AI kami dapat membantu dengan berbagai pertanyaan keuangan, termasuk tips penganggaran, strategi pengurangan pengeluaran, saran investasi, dan perencanaan keuangan umum. Ia menganalisis data keuangan Anda untuk memberikan rekomendasi yang dipersonalisasi.

Bagaimana riwayat transaksi saya disimpan?

Semua data transaksi disimpan secara lokal di browser Anda menggunakan metode penyimpanan yang aman. Ini memastikan informasi keuangan Anda tetap pribadi dan hanya dapat diakses oleh Anda. Anda dapat mengekspor data Anda kapan saja untuk tujuan pencadangan.

Bisakah saya menyesuaikan grafik keuangan?

Ya, grafik sepenuhnya dapat disesuaikan. Anda dapat memilih periode waktu, kategori, dan jenis visualisasi yang berbeda untuk lebih memahami pola keuangan Anda dan membuat keputusan yang tepat.
